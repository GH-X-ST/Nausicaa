\subsubsection{Pseudocode for Optimisation Workflow}\label{sec:E.1.1}

\vspace{0.30cm}

\begin{algorithm}[H]
\small
\setlength{\baselineskip}{1.15\baselineskip}
\caption{Nominal AeroSandbox~\cite{AeroSandbox} glider design optimisation}
\label{alg:E.1.1}
\KwIn{Design bounds \(\mathcal{B}\), fixed parameters \(\theta\), mission condition \(\xi_0\), objective \(J\), and requirements \(\mathcal{C}\).}
\KwOut{Feasible design \(d^\star\) and feasibility audit.}
Define the decision vector \(z=\{\alpha,\delta_a,\delta_e,\delta_r,b,c,l_t,b_H,b_V\}\) with \(z\in\mathcal{B}\)\;
Construct geometry \(G(z;\theta)\) and mass properties \(M(z;\theta)\)\;
Evaluate the nominal aerodynamic trim model at \(\xi_0\)\;
Compute feasibility margins and objectives listed in \textcolor{ICLBlue}{Tables~\ref{tab:E.1.2}} and \ref{tab:E.1.3}\;
Solve \(\min_z J(z)\) subject to \(z\in\mathcal{B}\) and \(\mathcal{C}(z)\leq0\)\;
\eIf{the solver converges and the post-solve audit passes}{
    Accept the solution as \(d^\star\)\;
}{
    Reject the candidate as infeasible\;
}
\textbf{return} \(d^\star\) or an infeasibility flag\;
\end{algorithm}

\vspace{0.70cm}

\begin{algorithm}[H]
\small
\setlength{\baselineskip}{1.15\baselineskip}
\caption{Multistart candidate selection and robust postcheck}
\label{alg:E.1.2}
\KwIn{Initial guesses \(\{z_0^{(i)}\}_{i=1}^{N_s}\), retained count \(K\), uncertainty scenarios \(\Omega\), and \textcolor{ICLBlue}{Algorithm~\ref{alg:E.1.1}}.}
\KwOut{Robust selected candidate \(d^\star\) and robust audit.}
\(\mathcal{F}\gets\emptyset\)\;
\For{each initial guess \(z_0^{(i)}\)}{
    Solve the nominal design problem from \(z_0^{(i)}\)\;
    \If{the solution is feasible}{
        Add the candidate and its nominal objective to \(\mathcal{F}\)\;
    }
}
\If{\(\mathcal{F}\) is empty}{
    \textbf{return} a failure flag\;
}
Sort \(\mathcal{F}\) by nominal objective, remove duplicate geometries, and retain the best \(K\) candidates\;
Sample uncertainty scenarios \(\{\omega_j\}_{j=1}^{N_\omega}\) from the intervals in \textcolor{ICLBlue}{Table~\ref{tab:E.1.5}}\;
\For{each retained candidate \(d_i\)}{
    \For{each scenario \(\omega_j\)}{
        Apply \(\omega_j\), re-solve the constrained trim problem, and evaluate the robust audit metrics\;
    }
    Summarise the success rate, tail sink metric, and worst-margin statistics for \(d_i\)\;
}
Rank candidates by success rate descending, tail mean sink rate and nominal objective ascending\;
\textbf{return} the highest ranked candidate \(d^\star\) and its robust audit\;
\end{algorithm}

\vspace{0.50cm}

\begin{algorithm}[H]
\small
\setlength{\baselineskip}{1.15\baselineskip}
\caption{Finite difference sensitivity analysis}
\label{alg:E.1.3}
\KwIn{Selected design \(d^\star\), parameter set \(\mathcal{P}\), output metrics \(\mathcal{Q}\), step ladder \(\mathcal{H}\), and stability tolerances.}
\KwOut{Sensitivity table \(\mathcal{S}\), selected step sizes, and stability flags.}
Reconstruct and re-evaluate the feasible baseline design to obtain \(q_j^0\) for all \(q_j\in\mathcal{Q}\)\;
\For{each parameter \(p_i\in\mathcal{P}\)}{
    Choose a central-difference scheme, or a forward scheme if at the lower bound\;
    Build the admissible step ladder \(h_{i,\ell}\in\mathcal{H}\)\;
    \For{each step \(h_{i,\ell}\)}{
        Evaluate the admissible perturbed designs and store their metrics and feasibility flags\;
    }
    \For{each output metric \(q_j\in\mathcal{Q}\)}{
        Form finite-difference derivative candidates from the feasible perturbations\;
        Use Richardson extrapolation and round-off estimates to select the nonround-off step\;
        \eIf{a defensible step is found}{
            Compute \(S_{ij}=(p_i/\max(|q_j^0|,q_{j,\min}))\,\partial q_j/\partial p_i\) and its stability flag\;
        }{
            Mark the sensitivity as unresolved\;
        }
    }
}
\end{algorithm}
